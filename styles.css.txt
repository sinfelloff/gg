// Mobile nav toggle
const toggle = document.getElementById("navToggle");
const links = document.getElementById("navLinks");

if (toggle && links) {
  toggle.addEventListener("click", () => {
    const open = links.classList.toggle("show");
    toggle.setAttribute("aria-expanded", open ? "true" : "false");
  });

  // Close menu when clicking a link
  links.querySelectorAll("a").forEach(a => {
    a.addEventListener("click", () => {
      links.classList.remove("show");
      toggle.setAttribute("aria-expanded", "false");
    });
  });
}

// Email signup (demo only)
const signupForm = document.getElementById("signupForm");
const signupNote = document.getElementById("signupNote");

if (signupForm && signupNote) {
  signupForm.addEventListener("submit", (e) => {
    e.preventDefault();
    const email = new FormData(signupForm).get("email");
    signupNote.textContent = `Thanks! We’ll keep you posted at ${email}. (Demo — connect to Mailchimp/Action Network later.)`;
    signupForm.reset();
  });
}

// Volunteer form (demo only)
const volunteerForm = document.getElementById("volunteerForm");
const volunteerNote = document.getElementById("volunteerNote");

if (volunteerForm && volunteerNote) {
  volunteerForm.addEventListener("submit", (e) => {
    e.preventDefault();
    const data = Object.fromEntries(new FormData(volunteerForm).entries());
    volunteerNote.textContent = `Thanks, ${data.name}! We’ll reach out soon. (Demo — wire this to Google Form/Action Network later.)`;
    volunteerForm.reset();
  });
}
